Hive 题目练习

No tag
Publish Date:   2021-10-16

1.统计 2020 年上映的电影中，当前总票房最高的 10 部电影。

create database mydb;

use mydb;

create table moviecleaned(
movie_name string,
boxoffice string,
box_rate string,
sessions string,
show_count_rate string,
avg_number string,
attendance string,
total_boxoffice string,
movie_days string,
current_time string,
releaseDate string
)row format delimited fields terminated by '\t';

load data local inpath "/data/workspace/myshixun/data/movies.txt" into table moviecleaned;

//相同的电影开窗，根据总票房排序
select *,row_number() 
over(partition by movie_name order by cast(total_boxoffice as double) desc) rn from moviecleaned where movie_name='八佰'

//取排名第一位的电影数据并且只取2020年的数据
select * from(select *,row_number() 
over(partition by movie_name order by cast(total_boxoffice as double) desc) rn from moviecleaned) t1 
where t1.rn=1 and releaseDate like '2020%';

//取排名之后的前十位
select movie_name,total_boxoffice from(select * from(select *,row_number() 
over(partition by movie_name order by cast(total_boxoffice as double) desc) rn from moviecleaned) t1 
where t1.rn=1 and releaseDate like '2020%') t2
order by cast(total_boxoffice as double) desc limit 10;

//把数据放入到top10_boxoffice表中
create table top10_boxoffice(
movie_name string,
total_boxoffice double
);
insert overwrite table top10_boxoffice select movie_name,total_boxoffice from(select * from(select *,row_number() 
over(partition by movie_name order by cast(total_boxoffice as double) desc) rn from moviecleaned) t1 
where t1.rn=1 and releaseDate like '2020%') t2
order by cast(total_boxoffice as double) desc limit 10;

2.在 hive 数据库 mydb 中，创建 boxoffice_national_day 表，使用 Hive SQL 来统计国庆假期中电影票房增长最多的三部电影及其每日的票房数据（2020 年国庆假期为 2020 年 10 月 1 日至 2020 年 10 月 7 日），并插入到boxoffice_national_day表。

//取增长最多的三部电影
select movie_name,sum(boxoffice) as n from moviecleaned WHERE current_time between '2020-10-01' and '2020-10-07' GROUP BY movie_name order by n desc LIMIT 3


//显示每日的票房数据
//方法一
select movie_name,boxoffice,current_time from moviecleaned WHERE movie_name in(select t.movie_name from(select movie_name,sum(boxoffice) as n from moviecleaned WHERE current_time between '2020-10-01' and '2020-10-07' GROUP BY movie_name order by n desc LIMIT 3) as t) and  current_time between '2020-10-01' and '2020-10-07';
//方法二
select a.movie_name,b.boxoffice,b.current_time from
(select movie_name,sum(boxoffice) as n from moviecleaned WHERE current_time between '2020-10-01' and '2020-10-07' GROUP BY movie_name order by n desc LIMIT 3) a join moviecleaned b on a.movie_name=b.movie_name and current_time between '2020-10-01' and '2020-10-07';


//把数据放入到top10_boxoffice表中
create table boxoffice_national_day    (
movie_name String,
boxoffice float,
dates string
);

insert overwrite table boxoffice_national_day select a.movie_name,b.boxoffice,b.current_time from
(select movie_name,sum(boxoffice) as n from moviecleaned WHERE current_time between '2020-10-01' and '2020-10-07' GROUP BY movie_name order by n desc LIMIT 3) a join moviecleaned b on a.movie_name=b.movie_name and current_time between '2020-10-01' and '2020-10-07';

3.在 hive 数据库 mydb 中，创建 day_max_boxoffice 表，使用 Hive SQL 来统计统计 2020 年中当日综合总票房最多的 10 天（以当日综合票房为统计依据，票房数据保留 2 位小数），并插入到day_max_boxoffice表。

//统计当日总票房
select current_time,sum(boxoffice) total from moviecleaned group by current_time order by total desc

//2020年当日总票房前十位，并且保留两位小数。
select current_time,round(total,2) totalboxoffice from(select current_time,sum(boxoffice) total from moviecleaned group by current_time order by total desc) t1
where current_time like '2020%' limit 10;

//把数据插入到day_max_boxoffice 表中
create table day_max_boxoffice(
dates string,
boxoffiice float
);
insert overwrite table day_max_boxoffice select current_time,round(total,2) totalboxoffice from(select current_time,sum(boxoffice) total from moviecleaned group by current_time order by total desc) t1
where current_time like '2020%' limit 10;

4.在 hive 数据库 mydb 中，创建 movie_boxoffice 表，使用 Hive SQL 来统计统计 2020 年首映的电影上映后 7 天的电影票房信息，并插入到movie_boxoffice表。

select movie_name,current_time,boxoffice from moviecleaned where current_time < date_add(releaseDate,7) and releaseDate like '2020%'
order by movie_name,current_time;

//把数据插入到 movie_boxoffice 表中
create table movie_boxoffice(
movie_name string,
dates string,
boxoffice float
);
insert overwrite table movie_boxoffice select movie_name,current_time,boxoffice from moviecleaned where current_time < date_add(releaseDate,7) and releaseDate like '2020%'
order by movie_name,current_time;

5.在 hive 数据库 mydb 中，创建 festival_boxoffice 表，使用 Hive SQL 来统计 2020 年元旦节与国庆节放假后 7 天的观影人数（元旦假期为 2020 年 1 月 1 日到 2020 年 1 月 7 日，国庆假期为 2020 年 10 月 1 日到 2020 年 10 月 7 日，计算规则为：观影人数=排片场次∗场均人次），并插入到festival_boxoffice 表。


//new_year_day   national_day

//用case when end 设置festival字段
select substr(current_time,-2,2),
case 
when current_time between '2020-10-01' and '2020-10-07' then 'new_year_day'
when current_time between '2020-01-01' and '2020-01-07' then 'national_day'
end as festival,
cast(sum(sessions*avg_number) as int) as num
from moviecleaned where current_time between '2020-10-01' and '2020-10-07' 
or 
current_time between '2020-01-01' and '2020-01-07'
group by current_time;

//把数据插入到 festival_boxoffice 表中
create table festival_boxoffice(
dates string,
festival string,
num bigint
);
insert overwrite table festival_boxoffice
select substr(current_time,-2,2),
case 
when current_time between '2020-10-01' and '2020-10-07' then 'new_year_day'
when current_time between '2020-01-01' and '2020-01-07' then 'national_day'
end as festival,
cast(sum(sessions*avg_number) as int) as num
from moviecleaned where current_time between '2020-10-01' and '2020-10-07' 
or 
current_time between '2020-01-01' and '2020-01-07'
group by current_time;
